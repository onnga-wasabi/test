from oauth2client.service_account import ServiceAccountCredentials


import json
import gspread                                                                                                                                                                                                                                
import os.path
import sys

if os.path.exists("list.txt"):
    sys.exit()

scope=['https://spreadsheets.google.com/feeds']
credentials=ServiceAccountCredentials.from_json_keyfile_name('spreadsheet.json',scope)
gc=gspread.authorize(credentials)
wks=gc.open_by_key('ID').sheet1

dict={}
key=[]
for i in range(2,7):
    key.append(wks.acell('A'+str(i)).value)
    dict[key[i-2]]=wks.acell('B'+str(i)).value 

flg=0
for i in dict.keys():
    if dict[i]!="":
        with open("list.txt","a") as file:
            file.write(dict[i]+"\n")

        if flg==0:
            list=dict[i].split("/")
            s=set(list)
            for mojiretu in s:
                if "v2_0" in mojiretu:
                    with open("tag.txt","a") as file2:
                        file2.write(mojiretu)
            flg=1
